  Database MySQL Set-up
  ===========================
  
  create database SaveO;

  use SaveO;

  CREATE USER 'SaveO'@'localhost' IDENTIFIED BY 'SaveORandomPassw@ord';

  GRANT ALL PRIVILEGES ON \*.\* TO 'SaveO'@'localhost';

  FLUSH PRIVILEGES;
